// 🔳 ⁡⁢⁣⁣ТЕКУЩАЯ ПАПКА И СОДЕРЖИМОЕ⁡
{
    pwd    ⁡⁣⁣ ⁡⁣⁣                ⁡⁣⁣⁢# Пример вывода: /c/Users/Alex⁡
    ls                      ⁡⁣⁣⁢# Показать файлы и папки в текущей директории⁡
    ls -la                  ⁡⁣⁣⁢# Показать всё, включая скрытые файлы (.git, .env и т.д.)⁡
}
// 🔳 ⁡⁢⁣⁣ПЕРЕХОД МЕЖДУ ДИСКАМИ И ПАПКАМИ⁡
{
    cd /c                   ⁡⁣⁣⁢# Перейти на диск C: (корень)⁡
    cd /d/projects          ⁡⁣⁣⁢# Перейти в папку на диске D:⁡
    cd ~                    ⁡⁣⁣⁢# Перейти в домашнюю папку (обычно /c/Users/ТвоёИмя)⁡
    cd                      ⁡⁣⁣⁢# Перейти в домашнюю папку (обычно /c/Users/ТвоёИмя)⁡
    cd /c/Users/Andr/app    ⁡⁣⁣⁢# Пример: перейти в проект React⁡
}
// 🔳 ⁡⁢⁣⁣НАВИГАЦИЯ ВНУТРИ ПАПОК⁡
{
    ls                      ⁡⁣⁣⁢# Oтображает список файлов и каталогов в текущем каталоге⁡
    ls -l                   ⁡⁣⁣⁡⁣⁣⁡⁣⁣⁢# Bыводит расширенную информацию: разрешения, владелец, размер, дату последнего
                            ⁡⁣⁣⁢# изменения.⁡⁡
    ls -a                   ⁡⁣⁣⁡⁣⁣⁢# Показывает скрытые файлы (названия которых начинаются с точки).⁡
    ls -al                  ⁡⁣⁣⁢# Форматированный список, включающий скрытые папки и файлы⁡
    cd ..                   ⁡⁣⁣⁢# Подняться на уровень выше (в родительскую папку)⁡
    cd -                    ⁡⁣⁣⁢# Вернуться в предыдущую папку⁡
    cd documents/work       ⁡⁣⁣⁢# Перейти в подпапку⁡
    cd ~/projects/app       ⁡⁣⁣⁢# Перейти в домашнюю папку и открыть проект⁡
    pwd                     ⁡⁣⁣⁢# Показать текущую директорию («где я?»)⁡
    cat file.txt            ⁡⁣⁣⁢# Вывести постранично file.txt⁡
}
// 🔳 ⁡⁢⁣⁣КОПИРОВАНИЕ ПЕРЕМЕЩЕНИЕ⁡
{
    cp file.txt /dir        ⁡⁣⁣⁢# Копирует файл file.txt в /dir.⁡
    cp -r /project /dir     ⁡⁣⁣⁢# Kопирует /project в /dir рекурсивно (включая вложенные каталоги).⁡
    mv file.txt file1.txt   ⁡⁣⁣⁢# Переименовывает file.txt в file1.txt и перемещает в /dir если прописать⁡
}
// 🔳 ⁡⁢⁣⁣СОЗДАНИЕ И УДАЛЕНИЕ ПАПОК⁡
{
    mkdir my-new-project    ⁡⁣⁣⁢# Создать папку⁡
    mkdir -p projects/      ⁡⁣⁣⁢# Создать вложенную структуру⁡
    touch file.txt          ⁡⁣⁣⁢# создаёт пустой файл или изменяет дату открытия или изменения файла.⁡
    rm file.txt             ⁡⁣⁣⁢# Удалить файл file.txt⁡
    rm -r project/          ⁡⁣⁣⁢# Удалить дерикторию с файлами⁡
    rm -f file.txt          ⁡⁣⁣⁢# Удалить файл без подтверждения⁡
    rm -rf project/         ⁡⁣⁣⁢# удалить директорию project/ без подтверждения⁡ 
    rmdir empty-folder      ⁡⁣⁣⁢# Удалить **пустую** папку⁡
    rm -rf old-project      ⁡⁣⁣⁢# Удалить папку **со всем содержимым** (осторожно!)⁡
    ⚠️ ⁡⁣⁣⁢rm -rf необратимо удаляет файлы! Убедись, что указал правильный путь.⁡
}
