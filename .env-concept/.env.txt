// 🔳 ⁡⁢⁣⁣СЕКРЕТНЫЕ ДАННЫЕ (API Keys, Passwords, DB URLs)⁡
{
    // ☑️ Где хранить:
    В файле .env в корне проекта.
    # .env
    DATABASE_URL=postgres://user:pass@localhost:5432/db
    API_SECRET_KEY=super_secret_123
    BACKEND_API_URL=https://api.mysite.com
    // ☑️ Как подключить:
    // ✅ Node.js (Backend): Используйте пакет dotenv
    {
        npm install dotenv
        В самом начале entry-файла (например, index.js)
        require('dotenv').config();
        // Или для ES Modules: import 'dotenv/config';
    }
    // ✅ Использование:
    {
        const dbUrl = process.env.DATABASE_URL;
        const apiKey = process.env.API_SECRET_KEY;
    }
    // ✅ Frontend (Vite, React, Next.js):
    {
        // ⚡ Vite: Переменные должны начинаться с VITE_. Доступ через import.meta.env.
        {
            const url = import.meta.env.VITE_BACKEND_API_URL;
        }
        // ⚡ Create React App: Переменные должны начинаться с REACT_APP_. Доступ через process.env.
        {
            const url = process.env.REACT_APP_BACKEND_API_URL;
        }
        // ⚡ Next.js: Переменные для клиента должны начинаться с NEXT_PUBLIC_:
        {
            const url = process.env.NEXT_PUBLIC_BACKEND_API_URL;
        }
    }
}
